ViewItem In-Place Redesign Guide

This guide changes the existing ViewItem screen gradually. It avoids full-screen YAML paste. It is based on docs/VIEWITEM_1_SPEC.md, the live yaml/ViewItem.yaml field bindings, and the proven approval-stepper formulas from yaml/Main_1.yaml.

Do not rebuild the owner people-picker or document type choice list. Keep the existing cmb_Owner and drpDocType logic. The design change is layout, panels, cards, warning banner, steppers, and link-card presentation.

Phase 6 - Save Formula

Replace the old SubmitForm(Form1) dependency with this Patch pattern once the standalone controls exist.

Set(varErrTitle, IsBlank(Trim(txtTitle.Text)));
Set(varErrOwner, IsBlank(cmb_Owner.Selected.Mail));
Set(varErrDocType, IsBlank(drpDocType.Selected.Value));
Set(varErrPublishDate, IsBlank(dpkPublishDate.SelectedDate));
If(
    varErrTitle || varErrOwner || varErrDocType || varErrPublishDate,
    false,
    If(
        !IsBlank(dpkPublishDate.SelectedDate) &&
        DateDiff(Today(), dpkPublishDate.SelectedDate, "Days") >= 0 &&
        DateDiff(Today(), dpkPublishDate.SelectedDate, "Days") < 10 &&
        !ckbHighPriority.Value,
        Notify("Release date is within 10 days. Tick High Priority to override, or pick a later date.", NotificationType.Warning),
        Set(varSaving, true);
        Set(
            varSavedRecord,
            Patch(
                'Policy Proof Tracker',
                If(IsBlank(varSelectedRelease.ID), Defaults('Policy Proof Tracker'), varSelectedRelease),
                {
                    Title: Trim(txtTitle.Text),
                    'Document Type': {Value: drpDocType.Selected.Value},
                    Owner: {
                        Claims: "i:0#.f|membership|" & cmb_Owner.Selected.Mail,
                        Department: "",
                        DisplayName: cmb_Owner.Selected.DisplayName,
                        Email: cmb_Owner.Selected.Mail,
                        JobTitle: "",
                        Picture: ""
                    },
                    'Planned Publish Date': dpkPublishDate.SelectedDate,
                    'High Priority': ckbHighPriority.Value,
                    'Reason for High Priority': txtReasonHighPri.Text,
                    'External consultation required': ckbExtConsult.Value,
                    'External consultation details': txtExtConsultDetails.Text,
                    'Top Item': ckbTopItem.Value,
                    'Title for Newsletter': txtNewsletterTitle.Text,
                    Audience: txtAudience.Text,
                    'Item summary': txtSummary.Text,
                    Remarks: txtRemarks.Text,
                    'Approved by Pillar Lead (Ready to Release)': ckbPillarLead.Value,
                    'Approved For Release By': {Value: Coalesce(radApproval.Selected.Value, "Not Yet Approved")},
                    Published: ckbPublished.Value,
                    'Published Date': If(ckbPublished.Value, Coalesce(dpkPublishedDate.SelectedDate, Today()), Blank()),
                    'Expiry Date': If(!IsBlank(dpkExpiryDate.SelectedDate), dpkExpiryDate.SelectedDate, Blank())
                }
            )
        );
        If(
            IsEmpty(Errors('Policy Proof Tracker')),
            Set(varSelectedRelease, varSavedRecord);
            Set(varSelectedItem, varSavedRecord);
            Set(varSelectPolicy, "Filled");
            Set(varParentSaved, true);
            Set(varUnsavedChanges, false);
            ClearCollect(colUnpublishedPolicies, Filter('Policy Proof Tracker', Published = false));
            ClearCollect(colLinks, Filter(PolicyLinks, PolicyTrackerSource.Id = varSavedRecord.ID));
            Notify("Saved successfully!", NotificationType.Success),
            Notify("Save failed. Please check your connection and try again.", NotificationType.Error)
        );
        Set(varSaving, false)
    )
)
